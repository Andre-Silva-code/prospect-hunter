# Story 1.4: Gemini Integration & Message Generation

## Story Statement

> **As a** user,
> **I want** to generate personalized Instagram DM messages for prospects using Google Gemini AI,
> **so that** I can send targeted, relevant outreach messages at scale without manual composition.

---

## Overview

This story integrates Google's Gemini API (1.5 Pro model) to enable AI-powered personalized message generation. Users provide prospect information (name, specialty, bio, recent posts), and the system generates 150-200 character direct messages tailored to each prospect. The implementation includes Redis caching for performance, graceful fallback messaging on API failures, retry logic with exponential backoff, and comprehensive security controls including prompt injection prevention.

**Key Deliverables:**

- ✅ Google Gemini API client configured with 1.5 Pro model
- ✅ POST `/api/gemini/generate-message` endpoint for message generation
- ✅ Prompt templates with dynamic prospect personalization
- ✅ Redis caching layer with 7-day TTL to prevent duplicate API calls
- ✅ Graceful fallback generic messages on API failures
- ✅ Retry logic with exponential backoff (3 attempts max)
- ✅ Message data persisted to Supabase `messages` table with RLS
- ✅ Prompt injection prevention and input sanitization
- ✅ Message generation completes in < 3 seconds per prospect
- ✅ Unit tests for API integration, caching, and fallback logic

#### Gemini Client Implementation

**File:** `lib/gemini-client.ts`

```typescript
import { GoogleGenerativeAI } from "@google/generative-ai";

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);
const model = genAI.getGenerativeModel({ model: "gemini-1.5-pro" });

const PROMPT_TEMPLATE = `You are a professional medical business development specialist.
Generate a short, personalized Instagram DM to: {prospect_name}

Prospect Details:
- Specialty: {specialty}
- Bio: {bio}
- Recent Activity: {recent_posts}

Requirements:
- 150-200 characters (short for DM)
- Professional yet friendly tone
- Mention specific detail from their profile
- NO generic templates, be specific
- NO links or mentions
- NO calls to action, just introduction

Output ONLY the message text, nothing else.`;

const MAX_RETRIES = 3;
const BACKOFF_MS = [2000, 4000, 8000];

export interface GenerateMessageInput {
  prospect_name: string;
  specialty: string;
  bio: string;
  recent_posts: string;
}

export interface GenerateMessageOutput {
  message: string;
  fallback_used: boolean;
}

function sanitizeInput(input: string): string {
  return input
    .slice(0, 500)
    .replace(/[{}<>]/g, "")
    .trim();
}

function buildPrompt(input: GenerateMessageInput): string {
  const sanitized = {
    prospect_name: sanitizeInput(input.prospect_name),
    specialty: sanitizeInput(input.specialty),
    bio: sanitizeInput(input.bio),
    recent_posts: sanitizeInput(input.recent_posts),
  };

  return PROMPT_TEMPLATE.replace("{prospect_name}", sanitized.prospect_name)
    .replace("{specialty}", sanitized.specialty)
    .replace("{bio}", sanitized.bio)
    .replace("{recent_posts}", sanitized.recent_posts);
}

export async function generateMessage(input: GenerateMessageInput): Promise<GenerateMessageOutput> {
  const prompt = buildPrompt(input);
  let lastError: Error | null = null;

  for (let attempt = 0; attempt < MAX_RETRIES; attempt++) {
    try {
      const result = await model.generateContent(prompt, {
        timeout: 30000, // 30 second timeout
      });

      const text = result.response.text();

      // Validate output length
      if (text.length > 250) {
        console.warn("Gemini output exceeded safety margin, truncating");
        return {
          message: text.slice(0, 200),
          fallback_used: false,
        };
      }

      return {
        message: text,
        fallback_used: false,
      };
    } catch (error) {
      lastError = error as Error;
      const statusCode = (error as any).status;

      // Don't retry on client errors
      if (statusCode === 400) break;

      // Exponential backoff
      if (attempt < MAX_RETRIES - 1) {
        await new Promise((resolve) => setTimeout(resolve, BACKOFF_MS[attempt]));
      }
    }
  }

  // Use fallback message
  console.warn("Message generation failed, using fallback", lastError);
  return {
    message: `Hi ${sanitizeInput(input.prospect_name)}, I'm reaching out because your work in ${sanitizeInput(input.specialty)} aligns with our mission. Would love to connect!`,
    fallback_used: true,
  };
}
```

#### Redis Client

**File:** `lib/redis.ts`

```typescript
import { createClient, RedisClientType } from "redis";

let client: RedisClientType | null = null;

export async function getRedisClient(): Promise<RedisClientType | null> {
  if (!process.env.REDIS_URL) {
    return null;
  }

  if (!client) {
    client = createClient({
      url: process.env.REDIS_URL,
    });

    client.on("error", (error) => {
      console.error("Redis client error:", error);
      client = null;
    });

    await client.connect();
  }

  return client;
}

export async function cacheMessage(
  key: string,
  message: string,
  ttlSeconds: number = 604800 // 7 days
): Promise<void> {
  try {
    const redis = await getRedisClient();
    if (!redis) return;

    await redis.setEx(key, ttlSeconds, message);
  } catch (error) {
    console.warn("Failed to cache message:", error);
  }
}

export async function getCachedMessage(key: string): Promise<string | null> {
  try {
    const redis = await getRedisClient();
    if (!redis) return null;

    return await redis.get(key);
  } catch (error) {
    console.warn("Failed to retrieve cached message:", error);
    return null;
  }
}
```

#### Message Service

**File:** `services/message.service.ts`

```typescript
import { supabaseClient } from "@/lib/supabase-client";
import { getCachedMessage, cacheMessage } from "@/lib/redis";
import { generateMessage, GenerateMessageInput } from "@/lib/gemini-client";
import type { Message } from "@/types/message";
import crypto from "crypto";

export async function generateAndSaveMessage(
  userId: string,
  prospectId: string,
  input: GenerateMessageInput
): Promise<Message> {
  // Check cache first
  const cacheKey = `gemini:message:${prospectId}`;
  const cachedMessage = await getCachedMessage(cacheKey);

  let content: string;
  let fallback_used = false;

  if (cachedMessage) {
    content = cachedMessage;
  } else {
    const result = await generateMessage(input);
    content = result.message;
    fallback_used = result.fallback_used;

    // Cache the message
    await cacheMessage(cacheKey, content);
  }

  // Save to database
  const { data, error } = await supabaseClient
    .from("messages")
    .insert([
      {
        user_id: userId,
        prospect_id: prospectId,
        content,
        fallback_used,
        status: "draft",
      },
    ])
    .select()
    .single();

  if (error) throw error;
  return data;
}
```
