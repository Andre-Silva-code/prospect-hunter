# Story 1.3: Apify Integration & Prospect Fetching

## Story Statement

> **As a** user,
> **I want** to search for prospects on Instagram by specialty and location using Apify's scraping actors,
> **so that** I can build a list of qualified leads for outreach with relevant prospect data (name, bio, followers, engagement rate).

---

## Overview

This story integrates Apify's REST API to enable automated prospect scraping from Instagram. Users can trigger searches filtering by medical specialty and location, and the system retrieves prospect profiles with biographical and engagement metrics. The implementation includes robust error handling with exponential backoff retry logic, database persistence with RLS isolation, and audit logging for compliance.

**Key Deliverables:**

- ✅ Apify REST client configured and authenticated
- ✅ POST `/api/apify/search` endpoint accepting filters
- ✅ Apify Actor triggered for Instagram profile scraping
- ✅ Prospect data extracted and validated
- ✅ Prospects persisted to Supabase `prospects` table with RLS
- ✅ Retry logic with exponential backoff (3 attempts max)
- ✅ Comprehensive error handling and logging
- ✅ Audit trail for all search operations
- ✅ Unit tests for search, error handling, and retries
- ✅ Performance verified (50 prospects in < 2 seconds)

#### Apify Client Implementation

**File:** `lib/apify-client.ts`

```typescript
import { ApifyClient } from "@apify/client";

const INSTAGRAM_ACTOR_ID = "your-actor-id-from-apify-store";
const MAX_RETRIES = 3;
const BACKOFF_MS = [2000, 4000, 8000]; // Exponential backoff

export const apifyClient = new ApifyClient({
  token: process.env.APIFY_TOKEN,
});

export async function searchInstagramProspects(
  specialty: string,
  location: string,
  limit: number,
  recentActivityDays: number
) {
  let lastError: Error | null = null;

  for (let attempt = 0; attempt < MAX_RETRIES; attempt++) {
    try {
      const run = await apifyClient.actor(INSTAGRAM_ACTOR_ID).call({
        specialty,
        location,
        limit,
        recentActivityDays,
      });

      const dataset = await apifyClient.dataset(run.defaultDatasetId).listItems();
      return dataset.items;
    } catch (error) {
      lastError = error as Error;
      const statusCode = (error as any).statusCode;

      // Don't retry on client errors
      if (statusCode === 400) throw error;

      // Exponential backoff
      if (attempt < MAX_RETRIES - 1) {
        await new Promise((resolve) => setTimeout(resolve, BACKOFF_MS[attempt]));
      }
    }
  }

  throw lastError;
}
```

#### Prospect Service

**File:** `services/prospect.service.ts`

```typescript
import { supabaseClient } from "@/lib/supabase-client";
import type { Prospect } from "@/types/prospect";

export async function insertProspects(
  userId: string,
  prospects: Partial<Prospect>[],
  searchTag?: string
): Promise<Prospect[]> {
  const normalizedProspects = prospects.map((p) => ({
    ...p,
    user_id: userId,
    instagram_handle: p.instagram_handle?.toLowerCase().trim(),
    instagram_url: p.instagram_url?.toLowerCase().trim(),
    search_tag: searchTag,
  }));

  const { data, error } = await supabaseClient
    .from("prospects")
    .insert(normalizedProspects)
    .select();

  if (error) throw error;
  return data;
}

export async function getUserProspects(
  userId: string,
  limit: number = 100,
  offset: number = 0
): Promise<Prospect[]> {
  const { data, error } = await supabaseClient
    .from("prospects")
    .select()
    .eq("user_id", userId)
    .order("created_at", { ascending: false })
    .range(offset, offset + limit - 1);

  if (error) throw error;
  return data || [];
}
```

import { supabaseClient } from "@/lib/supabase-client";
import type { AuditLog } from "@/types/audit-log";

export async function logAuditEvent(
  userId: string,
  action: string,
  resourceType: string,
  metadata: Record<string, any>,
  result: "success" | "failure" | "partial",
  resourceId?: string
): Promise<AuditLog> {

    .select()
    .single();

  if (error) throw error;
  return data;
}
